Breez SDK - Nodeless

Fetching the balance API docs

Once connected, the balance can be retrieved at any time.

Rust
let info = sdk.get_info().await?;
let balance_sat = info.wallet_info.balance_sat;
let pending_send_sat = info.wallet_info.pending_send_sat;
let pending_receive_sat = info.wallet_info.pending_receive_sat;
Swift
if let info = try? sdk.getInfo() {
    let balanceSat = info.walletInfo.balanceSat
    let pendingSendSat = info.walletInfo.pendingSendSat
    let pendingReceiveSat = info.walletInfo.pendingReceiveSat

    print(balanceSat)
    print(pendingSendSat)
    print(pendingReceiveSat)
}
Kotlin
try {
    val info = sdk.getInfo()
    val balanceSat = info?.walletInfo?.balanceSat
    val pendingSendSat = info?.walletInfo?.pendingSendSat
    val pendingReceiveSat = info?.walletInfo?.pendingReceiveSat
} catch (e: Exception) {
    // handle error
}

Go
if info, err := sdk.GetInfo(); err == nil {
    balanceSat := info.WalletInfo.BalanceSat
    pendingSendSat := info.WalletInfo.PendingSendSat
    pendingReceiveSat := info.WalletInfo.PendingReceiveSat
    log.Printf("Balance: %v sats", balanceSat)
    log.Printf("Pending: send %v sats, receive %v sats", pendingSendSat, pendingReceiveSat)
}
